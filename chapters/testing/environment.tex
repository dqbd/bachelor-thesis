    \hline
    RAM             & 128\,GB                                  \\
    \hline
    Driver          & 465.31                                   \\
    \hline
    OS              & Arch Linux (kernel 4.12.9, KPTI enabled) \\
    \hline
    Host compiler   & GCC g++ 11.1.0                           \\
    \hline
    Device compiler & nvcc v11.3.58                            \\
    \hline
  \end{tabular}
